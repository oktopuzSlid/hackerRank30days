def fizzBuzz(n):
    # Write your code here
    for x in range(1,n+1,1):
        if  x%3==0 and x%5==0:
            print("FizzBuzz")  
        
        elif x%5==0 and x%5!=3:
            print("Buzz")
    
        elif x%3==0 and x%3!=5:
            print("Fizz") 
     
        else:
            print(x)
    
if __name__ == '__main__':
    n = int(input().strip())

    fizzBuzz(n)
